### ðŸŸ¢ Cart State Management (Simple but Solid)

Using localStorage for simplicity, but with proper patterns:

```typescript
// src/hooks/useCart.ts
'use client';

import { useState, useEffect, useCallback } from 'react';

interface CartItem {
  productId: string;
  name: string;
  price: number;
  quantity: number;
  image: string;
}

interface CartState {
  items: CartItem[];
  total: number;
}

const CART_KEY = 'supplement-cart';

export function useCart() {
  const [cart, setCart] = useState<CartState>({ items: [], total: 0 });
  const [isLoaded, setIsLoaded] = useState(false);

  // Load cart from localStorage on mount
  useEffect(() => {
    const stored = localStorage.getItem(CART_KEY);
    if (stored) {
      try {
        setCart(JSON.parse(stored));
      } catch {
        localStorage.removeItem(CART_KEY);
      }
    }
    setIsLoaded(true);
  }, []);

  // Persist cart to localStorage
  useEffect(() => {
    if (isLoaded) {
      localStorage.setItem(CART_KEY, JSON.stringify(cart));
    }
  }, [cart, isLoaded]);

  const addItem = useCallback((item: Omit<CartItem, 'quantity'>) => {
    setCart(prev => {
      const existing = prev.items.find(i => i.productId === item.productId);
      
      if (existing) {
        return {
          items: prev.items.map(i =>
            i.productId === item.productId
              ? { ...i, quantity: i.quantity + 1 }
              : i
          ),
          total: prev.total + item.price,
        };
      }
      
      return {
        items: [...prev.items, { ...item, quantity: 1 }],
        total: prev.total + item.price,
      };
    });
  }, []);

  const removeItem = useCallback((productId: string) => {
    setCart(prev => {
      const item = prev.items.find(i => i.productId === productId);
      if (!item) return prev;
      
      return {
        items: prev.items.filter(i => i.productId !== productId),
        total: prev.total - (item.price * item.quantity),
      };
    });
  }, []);

  const updateQuantity = useCallback((productId: string, quantity: number) => {
    if (quantity < 1) return removeItem(productId);
    
    setCart(prev => {
      const item = prev.items.find(i => i.productId === productId);
      if (!item) return prev;
      
      const diff = quantity - item.quantity;
      
      return {
        items: prev.items.map(i =>
          i.productId === productId ? { ...i, quantity } : i
        ),
        total: prev.total + (diff * item.price),
      };
    });
  }, [removeItem]);

  const clearCart = useCallback(() => {
    setCart({ items: [], total: 0 });
  }, []);

  const itemCount = cart.items.reduce((sum, item) => sum + item.quantity, 0);

  return {
    items: cart.items,
    total: cart.total,
    itemCount,
    isLoaded,
    addItem,
    removeItem,
    updateQuantity,
    clearCart,
  };
}
```

**Future upgrade path:** When ready for server-side cart, replace this hook internals without changing the interface.

```typescript
// app/api/webhooks/stripe/route.ts
import { NextRequest, NextResponse } from 'next/server';
import Stripe from 'stripe';
import { getAdminDb } from '@/lib/firebase-admin';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);

export async function POST(request: NextRequest) {
  const body = await request.text();
  const signature = request.headers.get('stripe-signature')!;

  let event: Stripe.Event;

  try {
    event = stripe.webhooks.constructEvent(
      body,
      signature,
      process.env.STRIPE_WEBHOOK_SECRET!
    );
  } catch (error) {
    return NextResponse.json({ error: 'Invalid signature' }, { status: 400 });
  }

  if (event.type === 'checkout.session.completed') {
    const session = event.data.object as Stripe.Checkout.Session;
    
    // Create order in Firestore
    const db = getAdminDb();
    const orderRef = db.collection('orders').doc();

      id: orderRef.id,

      fulfillmentStatus: 'unfulfilled',
      total: session.amount_total,
```

```typescript
    { url: `${baseUrl}/products`, changeFrequency: 'daily', priority: 0.9 },
    { url: `${baseUrl}/about`, changeFrequency: 'monthly', priority: 0.7 },
    { url: `${baseUrl}/contact`, changeFrequency: 'monthly', priority: 0.7 },
    { url: `${baseUrl}/faq`, changeFrequency: 'weekly', priority: 0.7 },
    { url: `${baseUrl}/blog`, changeFrequency: 'daily', priority: 0.8 },
  ];
  
  // Dynamic product pages
  const productsSnapshot = await db.collection('products')
    .where('status', '==', 'active')
    .get();
  
  const productPages = productsSnapshot.docs.map(doc => ({
    url: `${baseUrl}/products/${doc.data().slug}`,
    lastModified: doc.data().updatedAt?.toDate(),
    changeFrequency: 'weekly' as const,
    priority: 0.8,
  }));
  
  // Dynamic category pages
  const categoriesSnapshot = await db.collection('categories')
    .where('status', '==', 'active')
    .get();
  
  const categoryPages = categoriesSnapshot.docs.map(doc => ({
    url: `${baseUrl}/categories/${doc.data().slug}`,
    lastModified: doc.data().updatedAt?.toDate(),
    changeFrequency: 'weekly' as const,
    priority: 0.85,
  }));
  
  // Dynamic blog pages
  const blogSnapshot = await db.collection('blog')
    .where('status', '==', 'published')
    .get();
  
  const blogPages = blogSnapshot.docs.map(doc => ({
    url: `${baseUrl}/blog/${doc.data().slug}`,
    lastModified: doc.data().updatedAt?.toDate(),
    changeFrequency: 'monthly' as const,
    priority: 0.7,
  }));
  
  return [...staticPages, ...productPages, ...categoryPages, ...blogPages];
}
```
